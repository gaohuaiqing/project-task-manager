```typescript
// 列折叠状态管理
const [collapsedColumns, setCollapsedColumns] = useState<Set<string>>(() => {
  const saved = localStorage.getItem('wbs-collapsed-columns');
  return saved ? new Set(JSON.parse(saved)) : new Set();
});

// 切换折叠
const toggleColumnCollapse = (columnKey: string) => {
  const newCollapsed = new Set(collapsedColumns);
  if (newCollapsed.has(columnKey)) {
    newCollapsed.delete(columnKey);
  } else {
    newCollapsed.add(columnKey);
  }
  setCollapsedColumns(newCollapsed);
};

// 持久化到 localStorage
useEffect(() => {
  localStorage.setItem('wbs-collapsed-columns', JSON.stringify([...collapsedColumns]));
}, [collapsedColumns]);
```

### UI组件
- **位置**: 工具栏右侧
- **按钮**: "列显示" + Columns图标 + 折叠数量徽章
- **下拉菜单**: 所有可折叠列的开关列表
- **折叠提示**: 按钮上显示已折叠列数 `{collapsedColumns.size}`
